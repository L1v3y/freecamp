#include <stdio.h>
#include <unistd.h>

int main(int argc, char *argv[])
{
    if (argc < 3) {
        printf("Usage: %s file_name num_lines\n", argv[0]);
        return 1;
    }

    printf("Transfer to execl() function \n");
    execl("/usr/bin/head", "head", "-n", argv[2], argv[1], NULL);
    printf("This line will not execute \n");
    return 0;
}

#include <stdio.h>
#include <unistd.h>

int main() {
    printf("Transfer to execv() function \n");

    char *args[] = {"head", "-n", "2", "f1", NULL};
    execv("/usr/bin/head", args);

    printf("This line will not execute\n");
    return 0;
}

#include <stdio.h>
#include <unistd.h>

int main() {
    printf("Transfer to execvp() function \n");

    char *args[] = {"head", "-n", "2", "f1", NULL};
    execvp("head", args);

    printf("This line will not execute\n");
    return 0;
}
